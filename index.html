const FOLDER_NAME = 'TokoPintar';

function getOrCreateFolder() {
  const folders = DriveApp.getFoldersByName(FOLDER_NAME);
  return folders.hasNext() ? folders.next() : DriveApp.createFolder(FOLDER_NAME);
}

// Ambil SATU file dengan nama tsb. Kalau ada duplikat lama, simpan
// yang paling baru dan buang sisanya otomatis (self-healing).
function getSingleFile(folder, filename) {
  const it = folder.getFilesByName(filename);
  let keep = null;
  const dupes = [];
  while (it.hasNext()) {
    const f = it.next();
    if (!keep) { keep = f; continue; }
    if (f.getLastUpdated() > keep.getLastUpdated()) { dupes.push(keep); keep = f; }
    else { dupes.push(f); }
  }
  dupes.forEach(f => f.setTrashed(true));
  return keep;
}

function doPost(e) {
  // Kunci eksekusi: request lain harus antre, tidak boleh jalan
  // bersamaan menyentuh file yang sama. Ini yang mencegah duplikat.
  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(30000);
    const data = JSON.parse(e.postData.contents);
    const folder = getOrCreateFolder();
    
    // === TEST KONEKSI ===
    if (data.action === 'test') {
      return json({ success: true, 
        message: 'Koneksi OK! Folder: ' + folder.getName(),
        email: Session.getActiveUser().getEmail() });
    }
    
    // === PUSH (simpan JSON ke Drive) ===
    if (data.action === 'push') {
      const filename = (data.key || 'all') + '.json';
      const content = JSON.stringify(data.payload, null, 2);
      
      const existing = getSingleFile(folder, filename);
      let file;
      if (existing) {
        // UPDATE isi file yang sudah ada, bukan hapus lalu buat baru
        existing.setContent(content);
        file = existing;
      } else {
        file = folder.createFile(filename, content, MimeType.PLAIN_TEXT);
      }
      return json({ success: true, fileId: file.getId(), 
        filename: filename, size: content.length });
    }
    
    // === PULL (baca JSON dari Drive) ===
    if (data.action === 'pull') {
      const filename = (data.key || 'all') + '.json';
      const file = getSingleFile(folder, filename);
      if (!file) {
        return json({ success: false, error: 'File ' + filename + ' belum ada di Drive' });
      }
      const content = file.getBlob().getDataAsString();
      return json({ success: true, payload: JSON.parse(content) });
    }

    // === DEDUPE (bersihkan manual file dobel yang sudah terlanjur ada) ===
    if (data.action === 'dedupe') {
      const files = folder.getFiles();
      const byName = {};
      let removed = 0;
      while (files.hasNext()) {
        const f = files.next();
        const n = f.getName();
        if (!byName[n]) { byName[n] = f; continue; }
        const cur = byName[n];
        if (f.getLastUpdated() > cur.getLastUpdated()) { cur.setTrashed(true); byName[n] = f; }
        else { f.setTrashed(true); }
        removed++;
      }
      return json({ success: true, removed: removed, message: removed + ' file duplikat dibersihkan.' });
    }
    
    return json({ success: false, error: 'Action tidak dikenal: ' + data.action });
    
  } catch(err) {
    return json({ success: false, error: err.toString() });
  } finally {
    lock.releaseLock();
  }
}

function doGet(e) {
  return json({ status: 'TokoPintar Cloud Bridge OK' });
}

function json(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
